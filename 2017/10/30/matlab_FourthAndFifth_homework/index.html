if sum((Kt_origin - Kt_fop).^2) > sum((Kt_origin - Kt_spline).^2)

% func_5_33.m
function y = func_5_33(t,y,eps,r,w)
L = 1 + eps*(sin(w*t+9*pi/8).^7);
dL_dt = 7*eps*w*(sin(w*t+9*pi/8).^6).*cos(w*t+9*pi/8);
y = [y(2);-(2./L.*dL_dt+r*L).*y(2)-1./L.*sin(y(1))];
